        ]
        [Level 2
            [Level 3, ldir={\faApple}]
            [\ldots]
        ]
        [Readme.md, ifile]
    ]
    [Level 1, idir
        [Level 2]
        [test.tex, file]
    ]
]
\end{fancydir}
